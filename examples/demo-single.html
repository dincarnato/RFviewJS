API usage

// Pass everything in a single load() call
viewer.load({
  sequence:  SEQ,
  structure: STR,

  // SHAPE values — one per base
  values: [0.1, NaN, 0.85, ...],
  showColors: true,
  colorMap: {
    type:     'discrete',
    min:      0,
    nanColor: '#999999',
    stops: [
      { value: 0.3, color: '#1f2328' },
      { value: 0.7, color: '#f5c518' },
      { value: 1.0, color: '#cc0000' },
    ],
  },

  // Pair annotations
  pairAnnotColorMap: [
    { key: 'covarying', color: '#22c55e' },
    { key: 'conserved', color: '#3b82f6' },
  ],
  pairAnnotations: [
    // j is inferred from structure
    { i: 3,  key: 'covarying' },
    { i: 13, key: 'conserved' },
    { i: 27, color: '#f97316' },
  ],
});
